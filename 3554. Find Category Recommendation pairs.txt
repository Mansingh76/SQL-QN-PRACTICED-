3554. Find Category Recommendation Pairs


WITH CTE AS
(SELECT DISTINCT user_id, category 
FROM ProductPurchases P JOIN ProductInfo I ON P.product_id = I.product_id),
CTE2 AS 
(SELECT C1.user_id, C1.category AS category1, C2.category AS category2 FROM CTE C1 JOIN CTE C2 
ON C1.user_id = C2.user_id AND C1.category < C2.category)
SELECT category1, category2, COUNT(DISTINCT user_id) as customer_count FROM CTE2
GROUP BY category1, category2
HAVING COUNT(DISTINCT user_id) >= 3
ORDER BY customer_count DESC, category1 ASC, category2 ASC;
